41  import io.leangen.geantyref.AnnotationFormatException;
42  import io.leangen.geantyref.TypeFactory;
43  
44  public class ProviderAwareObjectFactoryAggregatorTests {
45  
46    @Test
47    public void emptyProviderAwareObjectFactoryTest() {
48      ProviderAwareObjectFactoryAggregate aggregator = new ProviderAwareObjectFactoryAggregate();
49      assertThat(aggregator.getAllObjects(Object.class)).isEmpty();
50      assertThat(aggregator.getObject("bob", Class.class)).isNull();
51      assertThat(aggregator.getObject(null, "bob", Class.class)).isNull();
52      assertThat(aggregator.getObject(Class.class)).isNull();
53      assertThat(aggregator.getObject((Annotation) null, Class.class)).isNull();
54      assertThat(aggregator.getAllObjects((Annotation) null, Class.class)).isNull();
55    }
56    
57    
58    @Target({ ElementType.METHOD, ElementType.CONSTRUCTOR, ElementType.FIELD, ElementType.PARAMETER })
59    @Retention(RUNTIME)
60    @Documented
61    public @interface RightType {
62      /**
63       * example.
64       * @return example.
65       */
66      Class<? extends MethodInterceptor> implementedBy() default MethodInterceptor.class;
67    }
68    
69    @Test
70    public void providerAwareObjectFactoryTest() throws AnnotationFormatException {
71      Map<String, Object> initial = new HashMap<>();
72      initial.put("int", 1);
73      SimpleObjectFactory factory = new SimpleObjectFactory(initial);
74      ProviderAwareObjectFactoryAggregate aggregator = new ProviderAwareObjectFactoryAggregate();
75      RightType rightTypeNoValue = TypeFactory.annotation(RightType.class, new HashMap<String, Object>());
76      aggregator.register(rightTypeNoValue, factory);
77      assertThat(aggregator.getAllObjects(Object.class)).hasSize(1);
78      assertThat(aggregator.getAllObjects(CharSequence.class)).hasSize(0);
79      assertThat(aggregator.getAllObjects(rightTypeNoValue, CharSequence.class)).hasSize(0);
80      assertThat(aggregator.getAllObjects(rightTypeNoValue, Integer.class)).hasSize(1);
81      assertThat(aggregator.getObject("bob", Class.class)).isNull();
82      assertThat(aggregator.getObject(Class.class)).isNull();
83      assertThat(aggregator.getObject(Integer.class)).isEqualTo(1);
84      assertThat(aggregator.getObject(rightTypeNoValue, "bob", Integer.class)).isNull();
85      assertThat(aggregator.getObject(rightTypeNoValue, "int", Integer.class)).isEqualTo(1);
86      assertThat(aggregator.getObject(rightTypeNoValue, Integer.class)).isEqualTo(1);
87      assertThat(aggregator.getObject(rightTypeNoValue, String.class)).isNull();
88    }
