23  package org.ehoffman.aop.context;
24  
25  import static java.lang.annotation.RetentionPolicy.RUNTIME;
26  
27  import java.lang.annotation.Documented;
28  import java.lang.annotation.ElementType;
29  import java.lang.annotation.Repeatable;
30  import java.lang.annotation.Retention;
31  import java.lang.annotation.Target;
32  
33  import org.aopalliance.intercept.MethodInterceptor;
34  import org.ehoffman.advised.ObjectFactory;
35  import org.ehoffman.aop.context.internal.IoCContextAdvice;
36  
37  @Target({ ElementType.METHOD, ElementType.CONSTRUCTOR, ElementType.FIELD, ElementType.PARAMETER })
38  @Retention(RUNTIME)
39  @Documented
40  @Repeatable(IoCContexts.class)
41  public @interface IoCContext {
42  
43    /**
44     * Names can be used to identify specific ObjectFactories for method injection on a test, if no name is specified, then a default
45     * name will be used, and parameters of the test will fail if the reference an annotation with a name that does not exits.
46     * 
47     * @return the name used to look up entities in this context
48     */
49    String name() default "__default";
50  
51    /**
52     * Only meaningful when used as Parameter annotation. Specifies the name of an instance in a the object factory related to this
53     * annotation we desire to be passed to the test as input at runtime.
54     * 
55     * @return the name of an instance in an {@link IoCContext}
56     */
57    String instance() default "__default";
58  
59    /**
60     * Classes that define the context we wish to work with.
61     * 
62     * @return an array of classes used to build an IoCContext (the means of doing so is up to the {@link #objectFactoryClass()})
63     */
64    Class<?>[] classes() default {};
65  
66    /**
67     * Classes of object factory we wish to use, if none is specified a discovery of object factories will ensue, if only one is
68     * supported by the current classloader, it will be used.
69     * 
70     * @return the implementation of {@link ObjectFactory} we wish to use to build the context
71     */
72    Class<? extends ObjectFactory> objectFactoryClass() default ObjectFactory.class;
73  
74    /**
75     * {@link #implementedBy()} returns a Class that implements {@link org.aopalliance.intercept.MethodInterceptor}. This field will
76     * be accessed via reflection so the name must be exact. If the class also implements {@link java.io.Closeable} the
77     * {@link java.io.Closeable#close()} method will be called at the close of the global context.
78     * 
79     * @return the {@link MethodInterceptor} used to build the context, should never need to be changed.
80     */
81    Class<? extends MethodInterceptor> implementedBy() default IoCContextAdvice.class;
82  }
