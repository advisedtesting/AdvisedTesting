23  package org.advisedtesting.core.internal;
24  
25  import java.lang.annotation.Annotation;
26  import java.lang.reflect.Method;
27  import java.util.Arrays;
28  import java.util.Collections;
29  import java.util.Iterator;
30  import java.util.LinkedHashMap;
31  import java.util.Map;
32  import java.util.Map.Entry;
33  import java.util.function.Function;
34  
35  import org.advisedtesting.core.ObjectFactory;
36  
37  public class ProviderAwareObjectFactoryAggregate implements ObjectFactory {
38  
39    // registrar capability, used to build the default object factory....
40    private Map<Annotation, ObjectFactory> contexts = new LinkedHashMap<>();
41  
42    public void register(Annotation annotation, ObjectFactory objectFactory) {
43      contexts.put(annotation, objectFactory);
44    }
45  
46    @Override
47    public <T> T getObject(Class<T> type) {
48      for (Entry<Annotation, ObjectFactory> entry : contexts.entrySet()) {
49        T found = entry.getValue().getObject(type);
50        if (found != null) {
51          return found;
52        }
53      }
54      return null;
55    }
56  
57    @Override
58    public <T> T getObject(String name, Class<T> type) {
59      return contexts.entrySet().stream().map((entry) -> entry.getValue().getObject(name, type)).filter(o -> o != null).findFirst()
60              .orElseGet(() -> null);
61    }
62  
63    public <T> T getObject(Annotation annotation, Class<T> type) {
64      return invokeOnFoundObjectFactory(annotation, o -> o.getObject(type));
65    }
66  
67    public <T> T getObject(Annotation annotation, String name, Class<T> type) {
68      return invokeOnFoundObjectFactory(annotation, o -> o.getObject(name, type));
69    }
70  
71    public <T> Map<String, T> getAllObjects(Annotation annotation, Class<T> type) {
72      return invokeOnFoundObjectFactory(annotation, o -> o.getAllObjects(type));
73    }
74  
75    @SuppressWarnings("unchecked")
76    @Override
77    public <T> Map<String, T> getAllObjects(Class<T> type) {
78      for (Entry<Annotation, ObjectFactory> context : contexts.entrySet()) {
79        Map<String, T> matches = context.getValue().getAllObjects(type);
80        if (matches != null && matches.size() > 0) {
81          return matches;
82        }
83      }
84      return Collections.unmodifiableMap(Collections.EMPTY_MAP);
85    }
86  
87    public Object[] getArgumentsFor(Method method) {
88      Class<?>[] parameterTypes = method.getParameterTypes();
89      Annotation[][] annotations = method.getParameterAnnotations();
90      Object[] parameters = new Object[parameterTypes.length];
91      for (int i = 0; i < parameterTypes.length; i++) {
92        parameters[i] = getArgumentFor(parameterTypes[i], annotations[i]);
93      }
94      return parameters;
95    }
96  
97    private Object getArgumentFor(Class<?> argumentType, Annotation[] annotations) {
98      Object output = null;
99      Iterator<Annotation> iter = Arrays.asList(annotations).iterator();
100     while (iter.hasNext() && output == null) {
101       Annotation annotation = iter.next();
102       String requestedInstanceName = AdviceAnnotationEvaluator.getInstanceIfPresent(annotation);
103       if (requestedInstanceName != null && !"".equals(requestedInstanceName)) {
104         output = invokeOnFoundObjectFactory(annotation, o -> o.getObject(requestedInstanceName, argumentType));
105       } else {
106         output = invokeOnFoundObjectFactory(annotation, o -> o.getObject(argumentType));
107       }
108     }   
109     return output != null ? output : this.getObject(argumentType);
110   }
111 
112   private <T, X> X invokeOnFoundObjectFactory(Annotation annotation, Function<ObjectFactory, X> function) {
113     String desiredName = AdviceAnnotationEvaluator.getNameIfPresent(annotation);
114     for (Entry<Annotation, ObjectFactory> context : contexts.entrySet()) {
115       if (context.getKey().annotationType().isAssignableFrom(annotation.annotationType())
116               && (desiredName == null || desiredName.equals(AdviceAnnotationEvaluator.getNameIfPresent(context.getKey())))) {
117         return function.apply(context.getValue());
118       }
119     }
120     return null;
121   }
