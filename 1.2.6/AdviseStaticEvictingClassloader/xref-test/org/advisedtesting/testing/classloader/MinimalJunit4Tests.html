23  package org.advisedtesting.testing.classloader;
24  
25  import static org.assertj.core.api.Assertions.assertThat;
26  import static org.assertj.core.api.Assertions.assertThatThrownBy;
27  import static org.junit.Assert.fail;
28  
29  import java.io.IOException;
30  
31  import org.advisedtesting.classloader.EvictingStaticTransformer;
32  import org.advisedtesting.classloader.RestrictiveClassloader;
33  import org.advisedtesting.junit4.Junit4AopClassRunner;
34  import org.junit.Rule;
35  import org.junit.Test;
36  import org.junit.rules.TemporaryFolder;
37  import org.junit.runner.RunWith;
38  import org.springframework.instrument.classloading.ShadowingClassLoader;
39  
40  import test.classloader.data.ContainsStaticLiteralNonFinal;
41  import test.classloader.data.ContainsStaticUnsetVar;
42  import test.classloader.data.StaticInitBlockClass;
43  
44  @RunWith(Junit4AopClassRunner.class)
45  public class MinimalJunit4Tests {
46  
47    @Rule
48    public TemporaryFolder folder = new TemporaryFolder();
49    
50    @Test
51    public void testSimpleClassloaderChecks() throws ClassNotFoundException {
52      ShadowingClassLoader loader = new ShadowingClassLoader(this.getClass().getClassLoader());
53      loader.addTransformer(new EvictingStaticTransformer());
54      loader.loadClass(TestStaticInitializationEvictionJunit4.class.getName());
55      assertThatThrownBy(() -> loader.loadClass(ContainsStaticLiteralNonFinal.class.getName())).isInstanceOf(ClassFormatError.class);
56    }
57    
58    @Test
59    @RestrictiveClassloader
60    public void shoudlFailUsingAClassWithAStaticInit() throws IOException {
61      try {
62        assertThat(folder.newFolder()).isDirectory().canRead().canWrite();
63        new StaticInitBlockClass();
64        fail("Class should not have been loadable.");
65      } catch (ClassFormatError cfe) {
66        System.out.println("yup. bad class. life is good.");
67      }
68    }
69    
70    
71    @Test
72    @RestrictiveClassloader(delegatingPackagesSuppliers = TestPackageSupplier.class)
73    public void shoudlNotFailUsingAClassWithAStaticInit() throws IOException {
74      try {
75        assertThat(folder.newFolder()).isDirectory().canRead().canWrite();
76        new StaticInitBlockClass();
77      } catch (ClassFormatError cfe) {
78        fail("Class should be loadable");
79      }
80    }
81    
82    @Test(expected = ClassFormatError.class)
83    @RestrictiveClassloader
84    public void logsInitBlock() {
85      new StaticInitBlockClass();
86    }
87    
88    @Test(expected = ClassFormatError.class)
89    @RestrictiveClassloader
90    public void logsNonLiteral() {
91      new ContainsStaticUnsetVar();
92    }
93  }
