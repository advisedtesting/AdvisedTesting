25  import java.io.IOException;
26  import java.util.ArrayList;
27  import java.util.List;
28  import java.util.function.Function;

30  import org.springframework.asm.AnnotationVisitor;

40  public class ClassContainsStaticInitialization implements Function<String, List<String>> {
41  
42    private final int versionOpcode;
43    
44    private int getVersionOpcode() {
45      try {
46          if (Opcodes.class.getField("ASM7") != null) {
47            return Opcodes.ASM7;
48          }
49      } catch (NoSuchFieldException | SecurityException e) {
50          try {
51            if (Opcodes.class.getField("ASM6") != null) {
52              return Opcodes.ASM6;
53            }
54          } catch (NoSuchFieldException | SecurityException e1) {
55  	      try {
56  	        if (Opcodes.class.getField("ASM5") != null) {
57  	          return Opcodes.ASM5;
58  	        }
59  	      } catch (NoSuchFieldException | SecurityException e2) {
60  	        return Opcodes.ASM4;
61  	      }
62  	    }
63      }
64      return Opcodes.ASM4;
65    }
66    
67    public ClassContainsStaticInitialization() {
68      this.versionOpcode = getVersionOpcode();
69    }
70  
71    @Override
72    public List<String> apply(String className) {
73      ClassReader reader;
74      try {
75        reader = new ClassReader(className);
76        UnsafeClassVistor visitor = new UnsafeClassVistor(versionOpcode);
77        reader.accept(visitor, 0);
78        return visitor.getErrors();
79      } catch (IOException ioe) {
80        throw new IllegalArgumentException("Class is not readable " + className.replace('/', '.'), ioe);
81      }
82    }
83    
84    public List<String> apply(byte[] bytes) {
85      ClassReader reader;
86      reader = new ClassReader(bytes);
87      UnsafeClassVistor visitor = new UnsafeClassVistor(versionOpcode);
88      reader.accept(visitor, 0);
89      return visitor.getErrors();
90    }

92    /**
93     * Not thread safe.
94     * 
95     * @author rex
96     */
97    private static class UnsafeClassVistor extends ClassVisitor {
98      
99      private String className;
100     
101     private boolean isEnumeration = false; //enumerations can not avoid static member variables -- just make sure they are final.
102     
103     private final List<String> errors;
104     
105     public UnsafeClassVistor(int api) {
106       super(api);
107       errors = new ArrayList<>();
108     }
109     
110     @Override
111     public void visit(int version, int access, String name, String signature, String superName, String[] interfaces) {
112       className = name;
113       isEnumeration = isEnum(access);
114     }    
115     
116     private boolean isStaticFinalLiteral(int access, Object value) {
117       return isStatic(access) && isFinal(access) && value != null;
118     }
119     
120     private boolean isStaticFinalEnumeration(int access) {
121       return isStatic(access) && isFinal(access) && isEnumeration;
122     }
123     
124     private boolean isAssertionSupport(String name) {
125       return "$assertionsDisabled".equals(name);
126     }
127     
128     @Override
129     public FieldVisitor visitField(int access, String name, String desc, String signature, Object value) {
130       if (isStatic(access) 
131           && !isSynthetic(access)
132           && !(isStaticFinalEnumeration(access)
133                || isStaticFinalLiteral(access, value)
134                || isAssertionSupport(name))) {
135         addError("Disallowed static field with name \"" + name + "\"");
136       }
137       return super.visitField(access, name, desc, signature, value);
138     }
139     
140     public boolean isStatic(int access) {
141       return ((access & Opcodes.ACC_STATIC) != 0);
142     }
143 
144     public boolean isFinal(int access) {
145       return ((access & Opcodes.ACC_FINAL) != 0);
146     }
147     
148     public boolean isEnum(int access) {
149       return ((access & Opcodes.ACC_ENUM) != 0);
150     }
151     
152     public boolean isSynthetic(int access) {
153       return ((access & Opcodes.ACC_SYNTHETIC) != 0);
154     }
155     
156     @Override
157     public MethodVisitor visitMethod(int access, String name, 
158                               String desc, String signature, String[] exceptions) {
159       if (!isEnumeration && "<clinit>".equals(name)) {
160         return new UnsafeStaticInitVistor(this.api, this);
161       }
162       return super.visitMethod(access, name, desc, signature, exceptions);
163     }
164     
165     public List<String> getErrors() {
166       return errors;
167     }
168     
169     public void addError(String error) {
170       if (errors != null) {
171         this.errors.add(error + " on class: " + className.replace('/', '.'));
172       }
173     }
174     
175   }

177   /**
178    * <p>
179    * This methodVisitor evicts any method who's only operations
180    * are anything other than setting the class's static
181    * $assertionsDisabled with {@link Class#desiredAssertionStatus}'s
182    * result.
183    * </p>
184    * <p>
185    * If a class uses the assert keyword this static member variable
186    * is constructed and set in this way.
187    * </p>
188    * <p>
189    * Should only be used to validate &lt;cinit&gt; methods.
190    * </p>
191    * @author rex
192    */
193   public static class UnsafeStaticInitVistor extends MethodVisitor {
194 
195     private boolean shouldEvict = false;
196     
197     private final UnsafeClassVistor visitor;
198     
199     public UnsafeStaticInitVistor(int api, UnsafeClassVistor visitor) {
200       super(api);
201       this.visitor = visitor;
202     }
203     
204 
205     @Override
206     public void visitFieldInsn(int opcode, String owner, String name, String desc) {
207       shouldEvict = shouldEvict 
208               || opcode != Opcodes.PUTSTATIC
209               || !"$assertionsDisabled".equals(name);
210       super.visitFieldInsn(opcode, owner, name, desc);
211     }
212 
213     @Override
214     public void visitLocalVariable(String name, String desc, String signature, Label start, Label end, int index) {
215       shouldEvict = true;
216       super.visitLocalVariable(name, desc, signature, start, end, index);
217     }
218     
219     @Override
220     public void visitEnd() {
221       if (shouldEvict) {
222         visitor.addError("Disallowed <cinit> method (does more than enable the assert keyword)");
223       }
224     }
225 
226     @Override
227     public void visitTypeInsn(int opcode, String type) {
228       shouldEvict = true;
229       super.visitTypeInsn(opcode, type);
230     }
231 
232     @Override
233     public void visitMethodInsn(int opcode, String owner, String name, String desc, boolean itf) {
234       // TODO Auto-generated method stub
235       shouldEvict = shouldEvict 
236           || opcode != Opcodes.INVOKEVIRTUAL
237           || !"java/lang/Class".equals(owner)
238           || !"desiredAssertionStatus".equals(name)
239           || !"()Z".equals(desc);
240       super.visitMethodInsn(opcode, owner, name, desc, itf);
241     }
242 
243     @Override
244     public void visitInvokeDynamicInsn(String name, String desc, Handle bsm, Object... bsmArgs) {
245       shouldEvict = true;
246       super.visitInvokeDynamicInsn(name, desc, bsm, bsmArgs);
247     }
248 
249     @Override
250     public void visitTableSwitchInsn(int min, int max, Label dflt, Label... labels) {
251       shouldEvict = true;
252       super.visitTableSwitchInsn(min, max, dflt, labels);
253     }
254 
255     @Override
256     public void visitLookupSwitchInsn(Label dflt, int[] keys, Label[] labels) {
257       shouldEvict = true;
258       super.visitLookupSwitchInsn(dflt, keys, labels);
259     }
260 
261     @Override
262     public void visitMultiANewArrayInsn(String desc, int dims) {
263       shouldEvict = true;
264       super.visitMultiANewArrayInsn(desc, dims);
265     }
266 
267     @Override
268     public void visitTryCatchBlock(Label start, Label end, Label handler, String type) {
269       shouldEvict = true;
270       super.visitTryCatchBlock(start, end, handler, type);
271     }
272 
273     @Override
274     public AnnotationVisitor visitTryCatchAnnotation(int typeRef, TypePath typePath, String desc, boolean visible) {
275       shouldEvict = true;
276       return super.visitTryCatchAnnotation(typeRef, typePath, desc, visible);
277     }
278 
279     @Override
280     public AnnotationVisitor visitLocalVariableAnnotation(int typeRef, TypePath typePath, Label[] start, Label[] end, int[] index,
281             String desc, boolean visible) {
282       shouldEvict = true;
283       return super.visitLocalVariableAnnotation(typeRef, typePath, start, end, index, desc, visible);
284     }
285 
286     @Override
287     public void visitLineNumber(int line, Label start) {
288       super.visitLineNumber(line, start);
289     }
290     
291   }
