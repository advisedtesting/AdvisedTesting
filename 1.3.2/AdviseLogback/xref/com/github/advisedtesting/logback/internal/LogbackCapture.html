23  package com.github.advisedtesting.logback.internal;
24  
25  import static org.slf4j.Logger.ROOT_LOGGER_NAME;
26  
27  import java.io.ByteArrayOutputStream;
28  import java.io.OutputStream;
29  import java.io.UnsupportedEncodingException;
30  import java.nio.charset.Charset;
31  
32  import ch.qos.logback.classic.Logger;
33  import ch.qos.logback.classic.LoggerContext;
34  import ch.qos.logback.classic.encoder.PatternLayoutEncoder;
35  import ch.qos.logback.classic.spi.ILoggingEvent;
36  import ch.qos.logback.classic.util.ContextSelectorStaticBinder;
37  import ch.qos.logback.core.OutputStreamAppender;
38  import ch.qos.logback.core.encoder.Encoder;
39  
40  /**
41   * Temporarily captures Logback output (mostly useful for tests). Based on https://gist.github.com/olim7t/881318.
42   */
43  public class LogbackCapture {
44  
45    private static final ThreadLocal<LogbackCapture> INSTANCE = new ThreadLocal<>();
46  
47    private final ByteArrayOutputStream logs;
48    private final OutputStreamAppender<ILoggingEvent> appender;
49  
50    /**
51     * Start capturing.
52     */
53    public static void start() {
54      if (INSTANCE.get() != null) {
55        throw new IllegalStateException("already started");
56      }
57      INSTANCE.set(new LogbackCapture());
58    }
59  
60    /**
61     * Stop capturing and return the logs.
62     * 
63     * @return a String containing all logging that occurred during the test execution on it's thread.
64     */
65    public static String stop() {
66      final LogbackCapture instance = INSTANCE.get();
67      if (instance == null) {
68        throw new IllegalStateException("was not running");
69      }
70      final String result = instance.stopInstance();
71      INSTANCE.remove();
72      return result;
73    }
74  
75    private LogbackCapture() {
76      this.logs = new ByteArrayOutputStream(4096);
77      Encoder<ILoggingEvent> encoder = buildEncoder();
78      this.appender = buildAppender(encoder, logs);
79      Logger logger = getLogbackLogger();
80      logger.addAppender(appender);
81    }
82  
83    private String stopInstance() {
84      appender.stop();
85      try {
86        return logs.toString("UTF-16");
87      } catch (final UnsupportedEncodingException cantHappen) {
88        return null;
89      }
90    }
91  
92    private static Logger getLogbackLogger() {
93      return getContext()
94             .getLogger(ROOT_LOGGER_NAME);
95    }
96  
97    private static Encoder<ILoggingEvent> buildEncoder() {
98      final PatternLayoutEncoder encoder = new PatternLayoutEncoder();
99      encoder.setPattern("[%p] %m%n");
100     encoder.setCharset(Charset.forName("UTF-16"));
101     encoder.setContext(getContext());
102     encoder.start();
103     return encoder;
104   }
105 
106   private static OutputStreamAppender<ILoggingEvent> buildAppender(final Encoder<ILoggingEvent> encoder,
107           final OutputStream outputStream) {
108     final OutputStreamAppender<ILoggingEvent> appender = new OutputStreamAppender<>();
109     appender.setName("logcapture");
110     appender.setContext(getContext());
111     appender.setEncoder(encoder);
112     appender.setOutputStream(outputStream);
113     appender.start();
114     return appender;
115   }
116 
117   /**
118    * Given http://www.slf4j.org/codes.html#substituteLogger we occasionally need to retry getting the logging context on test's
119    * startup
120    * 
121    * @return the current {@link LoggerContext}.
122    * 
123    * @throws NullPointerException if the {@link LoggerContext} hasn't been set.
124    * 
125    * @throws RuntimeException with a cause of {@link InterruptedException} if the thread is interrupted while waiting for the
126    *           {@link LoggerContext} to be set.
127    */
128   private static LoggerContext getContext() {
129     return new Wait<LoggerContext>()
130             .on(() -> ContextSelectorStaticBinder.getSingleton().getContextSelector().getDefaultLoggerContext())
131             .trying(10)
132             .toComplete();
133   }
134 }
